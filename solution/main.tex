\begin{enumerate}

	\item 
	since
	\begin{equation}
	\mathbf{r}=\mathbf{f}(\mathbf{c},t)=\mathbf{g}(\mathbf{c})h(t)
	\end{equation}
	$\Rightarrow$
	\begin{align}
	\mathbf{g}(\mathbf{c})=\frac{\mathbf{r}}{ h(t)}
	\end{align}
	$\Rightarrow$
	\begin{align}
		\mathbf{c}=\mathbf{g}^{-1}(\frac{\mathbf{r}}{ h(t)})
	\end{align}
	so
	\begin{align}
	\mathbf{v}&=\frac{\partial{\mathbf{r}}}{\partial t}\\
				&=\mathbf{g}(\mathbf{c})\dot{h}(t)\\
				&=\mathbf{g}(\mathbf{\mathbf{g}^{-1}(\frac{\mathbf{r}}{ h(t)})})\dot{h}(t)
	\end{align}
	
	\item 
	\begin{align}
	g(c)=c\\
	h(t)=t^2
	\end{align}
	and it is easy to find
	\begin{equation}
	g^{-1}(c)=c
	\end{equation}
	\begin{align}
	v&=g(g^{-1}(\frac{x}{h(t)}))\dot{h}(t)\\
	&=g^{-1}(\frac{x}{h(t)})\dot{h}(t)\\
		&=\frac{x}{h(t)}\dot{h}(t)\\
		&=2t\frac{x}{t^2}\\
		&=2\frac{x}{t}\\
	\end{align}
	
	
	\item
	at time $t$ the $c$th element is at $f(c,t)$ ,
	so the temperature of the $c$th element is
	\begin{align}
	T=g(f(c,t),t)
	\end{align}
	so the variation rate is
	\begin{align}
	\frac{dT}{dt}&=\frac{\partial g}{\partial x}\frac{df}{dt}+\frac{\partial g}{\partial t}
	\end{align}
